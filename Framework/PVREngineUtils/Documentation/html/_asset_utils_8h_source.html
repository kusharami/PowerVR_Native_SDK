417 inline void createSingleBuffersFromMeshes(
418  GraphicsContext& context,
419  MeshIterator_ meshIter, MeshIterator_ meshIterEnd,
420  VboInsertIterator_ outVbos,
421  IboInsertIterator_ outIbos)
422 {
423  int i = 0;
424  while (meshIter != meshIterEnd)
425  {
426  size_t total = 0;

428  {
429  total += meshIter->getDataSize(ii);
430  }
431 
432  api::Buffer vbo = context->createBuffer((uint32)total, types::BufferBindingUse::VertexBuffer, true);
433  size_t current = 0;
434  for (size_t ii = 0; ii < meshIter->getNumDataElements(); ++ii)
435  {
436  vbo->update((const void*)meshIter->getData(uint32(ii)), (uint32)current,
437  (uint32)meshIter->getDataSize(uint32(ii)));
438  current += meshIter->getDataSize((uint32)ii);
439  }
440 
441  outVbos = vbo;
442  if (meshIter->getNumFaces())
443  {
444  api::Buffer ibo = context->createBuffer(meshIter->getFaces().getDataSize(), types::BufferBindingUse::IndexBuffer, true);
445  ibo->update((void*)meshIter->getFaces().getData(), 0, meshIter->getFaces().getDataSize());
446  outIbos = ibo;
447  }
448  else
449  {
450  outIbos = api::Buffer();
451  }

455  ++meshIter;

525 inline void create3dPlaneMesh(uint32 width, uint32 length, bool vertexAttribTex, bool vertexAttribNormal,
526  assets::Mesh& outMesh)
527 {
528  const float32 halfWidth = width * .5f;
529  const float32 halfLength = length * .5f;
530 
531  glm::vec3 normal[4]=
532  {
533  glm::vec3(0.0f,1.0f,0.0f),
534  glm::vec3(0.0f,1.0f,0.0f),
535  glm::vec3(0.0f,1.0f,0.0f),
536  glm::vec3(0.0f,1.0f,0.0f)
537  };
538 
539  glm::vec2 texCoord[4] =
540  {
541  glm::vec2(0.0f,1.0f),
542  glm::vec2(0.0f,0.0f),
543  glm::vec2(1.0f,0.0f),
544  glm::vec2(1.0f,1.0f),
545  };
546 
547  glm::vec3 pos[4] =
548  {
549  glm::vec3(-halfWidth,0.0f,-halfLength),
550  glm::vec3(-halfWidth,0.0f,halfLength),

552  glm::vec3(halfWidth,0.0f,-halfLength)
553  };
554 
555  uint32 indexData[] =
556  {
557  0,1,2,
558  0,2,3
559  };

561  pvr::float32 vertData[32];
562  uint32 offset = 0 ;
563 
564  for(uint32 i = 0; i < 4; ++i)
565  {
566  memcpy(&vertData[offset], &pos[i], sizeof(pos[i]));
567  offset += 3;
568  if(vertexAttribNormal)
569  {
570  memcpy(&vertData[offset], &normal[i], sizeof(normal[i]));
571  offset += 3;
572  }
