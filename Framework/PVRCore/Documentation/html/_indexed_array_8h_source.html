361  void compact()
362  {
363  //We can do that because the last remove() tears down all datastructures used.
364  if (!myindex.size())
365  {
366  return;
367  }
368  //First, make sure there is something to compact...
369  deleteditemlisttype_::iterator unused_spot = myDeletedItems.begin();
370  while (myDeletedItems.size() && unused_spot != myDeletedItems.end())
371  {
372  //Last item in the storage vector.
373  size_t last = mystorage.size();
374 
375  //1) Trim the end of the vector...
376  while (last-- && mystorage[last].isUnused)
377  {
378  mystorage.pop_back();
379  //Rinse, repeat. If size is zero, there is nothing to do.
380  }
381 
382  //Either the storage is empty
383  if (mystorage.empty())
384  {
385  myDeletedItems.clear();
386  //the rest should already be cleared...
387  //the loop will exit naturally
388  }
389  else //Or we can have find its last item!
390  {
391  //2)Trim any items from the end of the unused spots list that may have been trimmed off by the vector...
392  while (unused_spot != myDeletedItems.end() && *unused_spot >= last)
393  {
394  myDeletedItems.erase(unused_spot++);
395  }
396  //Any spots left?
397  if (unused_spot != myDeletedItems.end())
398  {
399  //Do the actual data movement. After all we've been through, we know that
400  //i. The last item of the vector is a valid item (guaranteed by 1)
401  //ii. The unused spot is not out of bounds of the vector
402  //iii.
403  //Copy by hand(as we have not defined a move assignment operator for compatibility reasons.
404  //Also, since the string does not throw, this makes easier to reason about exceptions.
405 #ifdef PVR_SUPPORT_MOVE_SEMANTICS
406  mystorage[*unused_spot].value = std::move(mystorage[last].value);
407  mystorage[*unused_spot].key = std::move(mystorage[last].key);
408 #else
409  mystorage[*unused_spot].value = mystorage[last].value;
410  mystorage[last].value.ValueType_::~ValueType_();
411  new(&mystorage[last].value) ValueType_();
412  mystorage[*unused_spot].key = mystorage[last].key;
413  mystorage[last].key.clear();
414 #endif
415  mystorage[*unused_spot].isUnused = false;
416  myindex[mystorage[*unused_spot].key] = *unused_spot;
417  mystorage.pop_back();
418  myDeletedItems.erase(unused_spot++);
419  }//else : No action needed - unused spots has been trimmed off completely, so no movement is possible, or necessary...

463  bool relocate(const IndexType_& key, size_t index)
464  {
465  typename maptype_::iterator found = myindex.find(key);
466  if (found == myindex.end()) { return false; }
467  size_t old_index = myindex[key];
468  if (index == old_index) { return true; } //No-op
469  if (index + 1 > mystorage.size()) // Storage not big enough.
470  {
471  //Grow, and mark unused all required items. Need to add all spots (But the last) to unusedspots.
472  size_t oldsize = mystorage.size();
473  mystorage.resize(index + 1);
474  for (size_t i = oldsize; i < index; ++i)
475  {
476  myDeletedItems.push_front(i);
477  mystorage[i].isUnused = 1;
478  }
479  mystorage.back() = mystorage[old_index];
480  removefromvector(old_index);
481  }
482  else if (mystorage[index].isUnused) // Lucky! Storage is big enough, and the item is not used!
483  {
484  deleteditemlisttype_::iterator place = std::find(myDeletedItems.begin(), myDeletedItems.end(), index);
485  assert(place != myDeletedItems.end()); // Shouldn't happen! Ever!
486  myDeletedItems.erase(place);
487  mystorage[index] = mystorage[old_index];
488  removefromvector(old_index);
489  }
490  else // Whoops! Space is already occupied. Swap with the old item!
491  {
492  myindex[mystorage[index].key] = old_index;
493  std::swap(mystorage[index], mystorage[old_index]);
494  }
495  myindex[key] = index;
496  return true;
497  }
