156  int populateValidVersions(const std::string& filename, IAssetProvider& assetProvider)
157  {
158  _filenames.clear();
159  _assetProvider = &assetProvider;
160 
161  //Case 1: The filename already refers to a specific API. In this case, as soon as it
162  //is found, stop. We won't bother with others.
163  std::string name, extension;
164  strings::getFileNameAndExtension(filename, name, extension);
165  for (int i = 1; i < (int)Api::Count; ++i)
166  {
167  if (strings::endsWith(name, apiCode(Api(i))))
168  {
169  Stream::ptr_type str = _assetProvider->getAssetStream(filename, false);
170  if (str.get())
171  {
172  setFileForApi(filename, Api(i));
173  return 1; //This file has a suffix, so it only has 1 api
174  }
175  return 0; //Filename has a suffix but was not found... Let's not go over the top here, assume this is failure.
176  }
177  }
178  int count = 0;
179 
180  //Being here means that the filename did not have an API suffix. Which is the main case: Load everything you find.
181  //So now do the main work: Test all possible Apis...
182  for (int i = 1; i < (int)Api::Count; ++i)
183  {
184  string file1 = name + "_" + apiCode(Api(i)) + (extension.size() ? "." + extension : "");
185  if (i == (int)Api::Vulkan) { file1 += ".spv"; }
186  Stream::ptr_type str = _assetProvider->getAssetStream(file1, false);
187  if (str.get())
188  {
189  setFileForApi(file1, Api(i));
190  ++count;
191  continue;
192  }
193  file1 = name + apiCode(Api(i)) + (extension.size() ? "." + extension : "");
194  str = _assetProvider->getAssetStream(file1, false);
195  if (str.get())
196  {
197  setFileForApi(file1, Api(i));
198  ++count;
199  }
200  }
201 
202  //Lastly: If we still cannot find a file, there is a last case: Legacy (the user is using a file without extension).
203  //In this case, we will load it as minimum version (ES2) and get it over with.
204  if (count == 0)
205  {
206  Stream::ptr_type str = _assetProvider->getAssetStream(filename, false);
207  if (str.get())
208  {
209  setFileForApi(filename, Api::OpenGLES2);
